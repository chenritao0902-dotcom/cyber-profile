    <script>

        /* --- 1. 性能优化的 Matrix 动画 --- */

        const canvas = document.getElementById('matrix-canvas');

        const ctx = canvas.getContext('2d');

        let width, height;

        let columns;

        const fontSize = 16;

        let rainDrops = [];

        let animationId;

        let isLowPerf = false;


        function initMatrix() {

            width = canvas.width = window.innerWidth;

            height = canvas.height = window.innerHeight;

            columns = Math.floor(width / fontSize);

            rainDrops = new Array(columns).fill(1);

        }


        const katakana = 'アァカサタナハマヤャラワガザダバパイィキシチニヒミリヰギジヂビピウゥクスツヌフムユュルグズブヅプエェケセテネヘメレヱゲゼデベペオォコソトノホモヨョロヲゴゾドボポ0123456789';

        const alphabet = katakana + 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789';


        // 核心优化：使用时间戳控制帧率 (FPS Throttling)

        let lastTime = 0;

        const fps = 24; // 锁定24帧，电影质感且省电

        const nextFrameTime = 1000 / fps;


        function animate(currentTime) {

            if (isLowPerf) return; // 低性能模式停止渲染


            animationId = requestAnimationFrame(animate);

            

            const deltaTime = currentTime - lastTime;

            if (deltaTime > nextFrameTime) {

                ctx.fillStyle = 'rgba(0, 0, 0, 0.05)';

                ctx.fillRect(0, 0, width, height);

                ctx.fillStyle = '#0F0';

                ctx.font = fontSize + 'px monospace';


                for(let i = 0; i < rainDrops.length; i++) {

                    const text = alphabet.charAt(Math.floor(Math.random() * alphabet.length));

                    ctx.fillText(text, i*fontSize, rainDrops[i]*fontSize);

                    

                    if(rainDrops[i]*fontSize > height && Math.random() > 0.975){

                        rainDrops[i] = 0;

                    }

                    rainDrops[i]++;

                }

                lastTime = currentTime - (deltaTime % nextFrameTime);

            }

        }


        /* --- 2. 低性能模式切换 --- */

        const perfBtn = document.getElementById('perf-toggle');

        

        function togglePerf() {

            isLowPerf = !isLowPerf;

            document.body.classList.toggle('low-perf');

            

            if (isLowPerf) {

                cancelAnimationFrame(animationId);

                ctx.clearRect(0, 0, width, height); // 清空画布

                perfBtn.innerText = "[GRAPHICS: LOW]";

                localStorage.setItem('perfMode', 'low');

            } else {

                animate(0);

                perfBtn.innerText = "[GRAPHICS: HIGH]";

                localStorage.setItem('perfMode', 'high');

            }

        }


        // 初始化检查

        initMatrix();

        // 读取用户偏好

        if(localStorage.getItem('perfMode') === 'low') {

            togglePerf(); 

        } else {

            animate(0);

        }

        

        window.addEventListener('resize', () => {

            initMatrix();

            // 重置后重新填充，避免黑屏

            if(!isLowPerf) {

               ctx.fillStyle = '#000';

               ctx.fillRect(0,0,width, height);

            }

        });


        /* --- 3. 鼠标光照效果 --- */

        const container = document.getElementById('container');

        document.addEventListener('mousemove', (e) => {

            if(window.innerWidth > 768 && !isLowPerf) { // 仅在桌面端且高性能模式下启用

                const rect = container.getBoundingClientRect();

                const x = e.clientX - rect.left;

                const y = e.clientY - rect.top;

                container.style.setProperty('--mouse-x', `${x}px`);

                container.style.setProperty('--mouse-y', `${y}px`);

            }

        });


        /* --- 4. 智能打字机 (带记忆功能) --- */

        const text = "系统: 检测到访客... 正在解密个人档案... 加载完成。我是你的数字向导。";

        const typeEl = document.getElementById('typing-text');

        

        function typeWriter() {

            // 如果已经看过动画，直接显示文本

            if (localStorage.getItem('introShown') === 'true') {

                typeEl.innerText = text;

                return;

            }


            let i = 0;

            typeEl.innerHTML = "";

            function typing() {

                if (i < text.length) {

                    typeEl.innerHTML += text.charAt(i);

                    i++;

                    setTimeout(typing, 40);

                } else {

                    // 播放完成后标记为已读

                    localStorage.setItem('introShown', 'true');

                }

            }

            typing();

        }

        typeWriter();
